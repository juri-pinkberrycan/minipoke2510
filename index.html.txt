  <script>
    function setSymptom(symptom) {
      document.getElementById("symptomInput").value = symptom;
      suggestRecipe();
    }

    function suggestRecipe() {
      const symptom = document.getElementById("symptomInput").value.trim();
      const result = document.getElementById("result");

      if (!symptom) {
        result.textContent = "不調を入力または選んでください。";
        return;
      }

      // 簡易レシピ例
      const recipes = {
        "疲れやすい": "黒豆とクコの実のスープがオススメ。体に元気を与えます。",
        "冷え性": "生姜とねぎのスープで体を芯から温めましょう。",
        "目の疲れ": "ブルーベリーと菊花茶が効果的です。",
        "胃腸の不調": "山薬と鶏肉のおかゆが胃を優しく整えます。",
        "肌荒れ": "白きくらげと梨のデザートで潤いを補給しましょう。",
        "ストレス": "百合根とナツメのスープで心を落ち着けましょう。",
        "不眠": "クコの実と竜眼の甘いスープでリラックスを。",
        "肩こり": "シナモンとみかんの皮入り茶が血行を促します。"
      };

      result.textContent = recipes[symptom] || `${symptom}に合う薬膳レシピを準備中です。`;
    }
  </script>
